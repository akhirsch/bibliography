\subsection{The Less-Nondeterminism Relation}
\label{sec:less-nonderm-def}

\begin{mathparpagebreakable}
  \infer[]{~}
  {\CtrlNone \maybelessnd \CtrlNone} \and

  \infer[]{~}
  {\CtrlNone \maybelessnd E} \and

  \infer[]{E_1 \lessnd E_2}
  {E_1 \maybelessnd E_2} \and

  \infer[]{~}
  {X \lessnd X} \and

  \infer[]{~}
  {\CtrlFail \lessnd \CtrlFail} \and

  \infer[]{~}
  {\Ret{e} \lessnd \Ret{e}} \and

  \infer[]{E_{1,1} \lessnd E_{2,1} \\
    E_{2,1} \lessnd E_{2,2}}
  {\CtrlSeq{E_{1,1}}{E_{2,1}} \lessnd \CtrlSeq{E_{2,1}}{E_{2,2}}} \and

  \infer[]{~}
  {\CtrlFun{X}{E} \lessnd \CtrlFun{X}{E}} \and

  \infer[]{~}
  {\CtrlRec{X}{E} \lessnd \CtrlRec{X}{E}} \and

  \infer[]{E_{1,1} \lessnd E_{2,1} \\
    E_{2,1} \lessnd E_{2,2}}
  {E_{1,1}~E_{2,1} \lessnd E_{2,1}~E_{2,2}} \and

  \infer[]{E_1 \lessnd E_2}
  {\SendTo{E_1}{\ell} \lessnd \SendTo{E_2}{\ell}} \and

  \infer[]{~}
  {\RecvFrom{\ell} \lessnd \RecvFrom{\ell}} \and

  \infer[]{E_1 \lessnd E_2}
  {\ChooseFor{d}{\ell}{E_1} \lessnd \ChooseFor{d}{\ell}{E_2}} \and

  \infer[]{\question E_{1,1} \lessnd \question E_{2,1} \\
    \question E_{2,1} \lessnd \question E_{2,2}}
  {\AllowChoice{\ell}{\question E_{1,1}}{\question E_{2,1}} \lessnd \AllowChoice{\ell}{\question E_{2,1}}{\question E_{2,2}}} \and

  \infer[]{E_{1,1} \lessnd E_{2,1} \\
    E_{2,1} \lessnd E_{2,2} \\
    E_{3,1} \lessnd E_{3,2}}
  {\CtrlITE{E_{1,1}}{E_{2,1}}{E_{3,1}} \lessnd \CtrlITE{E_{2,1}}{E_{2,2}}{E_{2,3}}} \and

  \infer[]{~}
  {\CtrlTFun{\alpha}{E} \lessnd \CtrlTFun{\alpha}{E}} \and

  \infer[]{E_1 \lessnd E_2}
  {E_1~t \lessnd E_2~t} \and

  \infer[]{E_{1,1} \lessnd E_{2,1} \\
    E_{2,1} \lessnd E_{2,2}}
  {\CtrlLetIn{\Ret{x}}{E_{1,1}}{E_{2,1}} \lessnd \CtrlLetIn{\Ret{x}}{E_{1,2}}{E_{2,2}}} \and

  \infer[]{E_{1,1} \lessnd E_{2,1} \\
    E_{2,1} \lessnd E_{2,2}}
  {\LetSendIn{\alpha \knd \kappa}{E_{1,1}}{\rho}{E_{2,1}} \lessnd \LetSendIn{\alpha \knd \kappa}{E_{1,2}}{\rho}{E_{2,2}}} \and

  \infer[]{E_1 \lessnd E_2}
  {\LetRecvIn{\alpha \knd *_e}{\ell}{E_1} \lessnd \LetRecvIn{\alpha \knd *_e}{\ell}{E_2}} \and

  \infer[]{E_{1,1} \lessnd E_{2,1} \\
    E_{2,1} \lessnd E_{2,2}}
  {\AmI{\ell}{E_{1,1}}{E_{2,1}} \lessnd \AmI{\ell}{E_{1,2}}{E_{2,2}}} \and
\end{mathparpagebreakable}
